29def check_sensors_for_obstacle(dx, dy, robot_name):
30 """
31 Checks the robot's sensors for obstacles in four directions and determines blocked movement directions.
32 Args:
33 dx (float): Desired movement in the X direction.
34 dy (float): Desired movement in the Y direction.
35 robot_name (str): The name of the robot, used to construct sensor signal names.
36 Returns:
37 tuple: (x_direction, y_direction)
38 x_direction (str): Indicates if movement in the X direction is blocked ('left', 'right', or 'Free').
39 y_direction (str): Indicates if movement in the Y direction is blocked ('front', 'back', or 'Free').
40 Notes:
41 - Uses global variable `latest_data` to access sensor readings.
42 - Prints a summary of detected obstacles and blocked directions.
43 """
44
45 sensor_directions = [
46 ('back', f"{robot_name}_S300_sensor1"),
47 ('left', f"{robot_name}_S300_sensor2"),
48 ('front', f"{robot_name}_S3001_sensor1"),
49 ('right', f"{robot_name}_S3001_sensor2"),
50 ]
51
52 obstacles = []
53
54 for direction, signal in sensor_directions:
55 points = latest_data.get(signal, [])
56 detected = None
57 for pt in points:
58 _, _, _, dist = pt
59 if dist:
60 detected = direction
61 break
62 obstacles.append(detected)
63
64 # Determine Y direction block
65 if obstacles[0] is not None and obstacles[2] is not None:
66 y_direction = 'front' if dy < 0 else 'back' if dy > 0 else "Free"
67 elif obstacles[0] is not None:
68 y_direction = 'back'
69 elif obstacles[2] is not None:
70 y_direction = 'front'
71 else:
72 y_direction = "Free"
73
74 # Determine X direction block
75 if obstacles[1] is not None and obstacles[3] is not None:
76 x_direction = 'right' if dx < 0 else 'left' if dx > 0 else "Free"
77 elif obstacles[1] is not None:
78 x_direction = 'left'
79 elif obstacles[3] is not None:
80 x_direction = 'right'
81 else:
82 x_direction = "Free"
83
84 print(f"🚨 [{robot_name}] Sensor obstacle check: {obstacles} | X dir: {x_direction}, Y dir: {y_direction}")
85 return x_direction, y_direction
