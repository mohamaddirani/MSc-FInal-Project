66 def computeTargetVelocity(jointAngle,targetAngle):
67 dynStepSize=0.005
68 velUpperLimit=360*math.pi/180
69 PID_P=0.1
70 errorValue=targetAngle-jointAngle
71 sinAngle=math.sin(errorValue)
72 cosAngle=math.cos(errorValue)
73 errorValue=math.atan2(sinAngle,cosAngle)
74 ctrl=errorValue*PID_P
75
76 # Calculate the velocity needed to reach the position in one dynamic time step:
77 velocity=ctrl/dynStepSize
78 if (velocity>velUpperLimit):
79 velocity=velUpperLimit
80
81 if (velocity<-velUpperLimit):
82 velocity=-velUpperLimit
83
84 return velocity
