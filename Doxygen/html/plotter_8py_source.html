6def plot_sensor_data(robot_name, filename="sensor_data.png"):
7 """
8 Plots LiDAR sensor data for a specified robot and saves the plot as an image file.
9 Parameters:
10 robot_name (str): The name of the robot whose sensor data will be plotted.
11 filename (str, optional): The filename for saving the plot image. Defaults to "sensor_data.png".
12 The function expects a global dictionary `all_sensor_data` where keys are sensor names and values are lists of (x, y) tuples representing sensor readings.
13 Each sensor is plotted with a distinct color and labeled accordingly.
14 """
15
16 plt.figure(figsize=(8, 8))
17
18 colors = {
19 'S300_sensor1': 'purple', # back
20 'S300_sensor2': 'orange', # left
21 'S3001_sensor1': 'red', # front
22 'S3001_sensor2': 'green', # right
23 }
24
25 for signal_suffix, color in colors.items():
26 signal = f"{robot_name}_{signal_suffix}"
27 points = all_sensor_data.get(signal, [])
28 xs = [pt[0] for pt in points]
29 ys = [pt[1] for pt in points]
30 plt.scatter(xs, ys, s=10, label=signal_suffix, color=color)
31
32 plt.title(f"LiDAR Sensor Data: {robot_name}")
33 plt.xlabel("X (m)")
34 plt.ylabel("Y (m)")
35 plt.grid(True)
36 plt.legend()
37 plt.savefig(filename, dpi=300)
38 #print(f"✅ Sensor data saved as: {filename}")
